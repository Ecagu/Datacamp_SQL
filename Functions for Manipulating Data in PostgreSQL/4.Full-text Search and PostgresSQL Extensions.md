### (1)
SELECT *
FROM film
WHERE title LIKE 'GOLD%';

### (2)
SELECT *
FROM film
WHERE title LIKE '%GOLD';

# What is a tsvector?
SELECT to_tsvector(description)
FROM film;

# Basic full-text search
SELECT title, description
FROM film
WHERE to_tsvector(title) @@ to_tsquery('elf');
